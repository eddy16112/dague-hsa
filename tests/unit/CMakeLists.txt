if( MPI_FOUND )
    set(lifo_CFLAGS  "${MPI_COMPILE_FLAGS} ${lifo_CFLAGS} -DUSE_MPI")
    set(lifo_LDFLAGS "${MPI_LINK_FLAGS} ${lifo_LDFLAGS}")
    set(lifo_LIBS    dague-mpi ${lifo_LIBS} ${MPI_LIBRARIES})
    set(list_CFLAGS  "${MPI_COMPILE_FLAGS} ${list_CFLAGS} -DUSE_MPI")
    set(list_LDFLAGS "${MPI_LINK_FLAGS} ${list_LDFLAGS}")
    set(list_LIBS    dague-mpi ${list_LIBS} ${MPI_LIBRARIES})
else ( MPI_FOUND )
    set(lifo_LIBS    dague ${lifo_LIBS})
    set(list_LIBS    dague ${list_LIBS})
endif()

add_executable(lifo lifo.c)
set_target_properties(lifo PROPERTIES COMPILE_FLAGS "-DADD_")
set_target_properties(lifo PROPERTIES LINKER_LANGUAGE C)
set_target_properties(lifo PROPERTIES LINK_FLAGS "${LOCAL_C_LINK_FLAGS}")
target_link_libraries(lifo ${lifo_LIBS} -lpthread ${EXTRA_LIBS} -lm)

add_executable(list list.c)
set_target_properties(list PROPERTIES COMPILE_FLAGS "-DADD_")
set_target_properties(list PROPERTIES LINKER_LANGUAGE C)
set_target_properties(list PROPERTIES LINK_FLAGS "${LOCAL_C_LINK_FLAGS}")
target_link_libraries(list ${list_LIBS} -lpthread ${EXTRA_LIBS} -lm)
